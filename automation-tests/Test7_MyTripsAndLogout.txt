package org.example;

import org.openqa.selenium.By;
import org.openqa.selenium.Dimension;
import org.openqa.selenium.JavascriptExecutor;
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.WebElement;
import org.openqa.selenium.chrome.ChromeDriver;
import org.openqa.selenium.support.ui.Select;
import org.openqa.selenium.support.ui.WebDriverWait;
import org.openqa.selenium.support.ui.ExpectedConditions;
import org.openqa.selenium.support.ui.WebDriverWait;

import java.time.Duration;

public class Test7_MyTripsAndLogout {

    static final String BASE_URL = "http://localhost:8000";

    public static void main(String[] args) throws InterruptedException {

        System.setProperty("webdriver.chrome.driver", "/usr/bin/chromedriver");

        WebDriver driver = new ChromeDriver();
        WebDriverWait wait = new WebDriverWait(driver, Duration.ofSeconds(15));
        JavascriptExecutor js = (JavascriptExecutor) driver;

        try {
            // ─────────────────────────────────────────
            // PREREQUISITE: Register, Login & Create a Trip
            // ─────────────────────────────────────────
            System.out.println("Setting up session + creating a trip...");
            driver.get(BASE_URL + "/register.html");
            wait.until(ExpectedConditions.titleContains("Register"));
            Thread.sleep(2000);

            String testEmail = "selenium_mytrips_" + System.currentTimeMillis() + "@trippilot.com";
            driver.findElement(By.id("reg-name")).sendKeys("MyTrips Tester");
            driver.findElement(By.id("reg-email")).sendKeys(testEmail);
            driver.findElement(By.id("reg-password")).sendKeys("Test@123!");
            driver.findElement(By.id("reg-confirm")).sendKeys("Test@123!");
            driver.findElement(By.id("register-form")).submit();
            wait.until(ExpectedConditions.urlContains("login.html"));

            driver.findElement(By.id("login-email")).sendKeys(testEmail);
            driver.findElement(By.id("login-pw")).sendKeys("Test@123!");
            driver.findElement(By.id("login-button")).click();
            wait.until(ExpectedConditions.urlContains("dashboard.html"));
            Thread.sleep(1500);

            // Create a trip so My Trips has data
            driver.navigate().to(BASE_URL + "/create-trip.html");
            wait.until(ExpectedConditions.presenceOfElementLocated(By.id("create-trip-form")));
            Thread.sleep(1500);
            driver.findElement(By.id("trip-name")).sendKeys("Summer Vacation in Tokyo");
            driver.findElement(By.id("trip-destination")).sendKeys("Tokyo, Japan");
            js.executeScript("document.getElementById('trip-start').value = '2026-07-10'");
            js.executeScript("document.getElementById('trip-end').value = '2026-07-20'");
            driver.findElement(By.id("trip-budget")).sendKeys("4500");
            driver.findElement(By.id("trip-travelers")).sendKeys("2");
            new Select(driver.findElement(By.id("trip-style"))).selectByValue("adventure");
            driver.findElement(By.id("generate-itinerary")).click();
            wait.until(ExpectedConditions.urlContains("itinerary.html"));
            Thread.sleep(2000);
            System.out.println("Trip created ✅, on itinerary: " + driver.getCurrentUrl());

            // ─────────────────────────────────────────
            // TEST 8: My Trips
            // ─────────────────────────────────────────
            System.out.println("\n===== TEST 8: My Trips =====");
            driver.navigate().to(BASE_URL + "/my-trips.html");
            wait.until(ExpectedConditions.titleContains("My Trips"));
            Thread.sleep(3000);

            System.out.println("Title       : " + driver.getTitle());
            System.out.println("Current URL : " + driver.getCurrentUrl());

            WebElement tripsContainer = driver.findElement(By.id("trips-container"));
            String tripsText = tripsContainer.getText();
            System.out.println("Trip list visible: " + (!tripsText.isEmpty() ? "YES ✅" : "NO ❌"));

            if (tripsText.contains("Summer Vacation in Tokyo")) {
                System.out.println("✅ Created trip found in My Trips list.");
            } else {
                System.out.println("❌ Created trip missing from My Trips.");
            }
            Thread.sleep(3000);

            // ─────────────────────────────────────────
            // TEST 9: Logout
            // ─────────────────────────────────────────
            System.out.println("\n===== TEST 9: Logout =====");
            wait.until(ExpectedConditions.presenceOfElementLocated(By.id("logout-btn")));
            driver.findElement(By.id("logout-btn")).click();
            Thread.sleep(3000);

            String userAfterLogout = (String) js.executeScript("return localStorage.getItem('tripPilotUser');");
            System.out.println("Session after logout : " + userAfterLogout);   // should be null
            System.out.println("After Logout URL     : " + driver.getCurrentUrl());

            if (userAfterLogout == null) {
                System.out.println("✅ Session cleared — Logged out successfully");
            } else {
                System.out.println("❌ Session still present after logout!");
            }

            System.out.println("\n===== TESTS 8 & 9 COMPLETED =====");
            Thread.sleep(3000);
            driver.quit();

        } catch (Exception e) {
            System.err.println("\n❌ TEST SUITE FAILED!");
            e.printStackTrace();
            driver.quit();
        }
    }
}
